<script>
document.addEventListener('DOMContentLoaded', function() {
  const grid = document.getElementById('gallery-grid');
  grid.innerHTML = '';
  
  const examples = [
    {% for example in site.gallery %}
    {
      title: "{{ example.title | escape }}",
      url: "{{ example.url | relative_url }}",
      language: "{{ example.language }}",
      languageIcon: "{{ example.language_icon }}",
      protocol: "{{ example.protocol }}",
      schema: "{{ example.schema_format }}",
      role: "{{ example.role }}",
      definition: "{{ example.definition_name }}"
    }{% unless forloop.last %},{% endunless %}
    {% endfor %}
  ];
  
  examples.forEach(function(ex) {
    const card = document.createElement('a');
    card.className = 'gallery-card';
    card.href = ex.url;
    card.dataset.language = ex.language.toLowerCase();
    card.dataset.protocol = ex.protocol.toLowerCase();
    card.dataset.schema = ex.schema.toLowerCase().replace(' ', '');
    
    card.innerHTML = `
      <div class="card">
        <div class="gallery-card-header">
          <i class="gallery-card-icon devicon-${ex.languageIcon}-plain"></i>
          <div class="gallery-card-tags">
            <span class="tag protocol">${ex.protocol}</span>
            <span class="tag schema-format">${ex.schema}</span>
          </div>
        </div>
        <h3 class="gallery-card-title">${ex.title}</h3>
        <div class="gallery-card-meta">
          <span>${ex.language}</span>
          <span class="role-badge ${ex.role.toLowerCase()}">${ex.role}</span>
        </div>
      </div>
    `;
    
    grid.appendChild(card);
  });
});
</script>
